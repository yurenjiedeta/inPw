- printf

```c
main(){
	printf("%3d %6d\n",fahr,celsius); // %3d 和 %6d，的 3 和 6 是输出占位符。
    printf("%3.0f %6.1f\n",fahr,celsius); // %3.0f 表示至少占 3 个字符宽，且不带小数点和小数部分
    // %6.1f 表示至少占 6 个字符宽，切小数点后面有一位数字
    
    // %.2f 表示小数点后有两位小数，没有宽度限制
    // %f 表示 按照浮点数打印该数
}
```
